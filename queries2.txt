FROM `students`

SELECT `office_address` AS `indirizzo_edificio` ,  COUNT(*) AS `numero_di_insegnanti`
FROM `teachers`
GROUP BY `indirizzo_edificio`;

3. Calcolare la media dei voti di ogni appello d'esame

SELECT `exam_id`, AVG(`vote`) AS `media_voti`
FROM `exam_student`
GROUP BY `exam_id`;
